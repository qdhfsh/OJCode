```c++
#include<iostream>
using namespace std;
int arr[105][105];
int dp[105][105]; 
int dir[4][2] = {{-1,0},{1,0},{0,-1},{0,1}};
int R,C;

bool judge(int x, int y){
	if(x < 1 || x > R || y < 1 || y > C)
		return false;
	return true;
}

int dfs(int i, int j){
	if(dp[i][j])
		return dp[i][j];
	int ret = 1;
	for(int k = 0; k < 4; k++){
		int x = i + dir[k][0];
		int y = j + dir[k][1];
		if(judge(x,y) && arr[i][j] > arr[x][y])
			ret = max(dfs(x,y)+1, ret);
	}
	dp[i][j] = ret;
	return ret;
}

int main(){
	cin>>R>>C; 
	for(int i = 1; i <= R; i++){
		for(int j = 1; j <= C; j++){
			cin>>arr[i][j];
		}
	}
	int ans = 0;
	for(int i = 1; i <= R; i++){
		for(int j = 1; j <= C; j++){
			ans = max(dfs(i,j),ans);
		}
	}
	cout<<ans;
	
	return 0;
}
```
