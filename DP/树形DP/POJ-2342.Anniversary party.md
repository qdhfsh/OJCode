```c++
#include<iostream>
#include<vector>

using namespace std;

const int N = 6010;

int n, a, b, root;
int val[N], dp[2][N], vis[N];
vector<int>son[N];

void dfs(int x){
	dp[0][x] = 0;
	dp[1][x] = val[x];
	for(int i = 0; i < son[x].size(); i++){
		int y = son[x][i];
		dfs(y);
		dp[0][x] += max(dp[0][y], dp[1][y]);
		dp[1][x] += dp[0][y];
	}
}

int main(){
	cin>>n;
	for(int i = 1; i <= n; i++){
		cin>>val[i];
	}
	
	while(cin>>a>>b && a != 0 && b != 0){
		vis[a] = 1; //有父节点做标记
		son[b].push_back(a);	
	}
	for(int i = 1; i <= n; i++){
		if(!vis[i]){
			root = i; break;
		}
	}
	dfs(root);
	cout<<max(dp[0][root], dp[1][root])<<endl;
	
	return 0;
}
```
